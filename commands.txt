/////////////////////////////////
// INSTALL DEPENDENCIES
/////////////////////////////////

Install Python 3.9 (other 3.x versions may also work, but not guaranteed)

/////////////////////////////////
// CREATE VENV AND INSTALL LIBS
/////////////////////////////////

cd backend
py -3 -m venv venv
venv\Scripts\activate
pip install Flask
pip install flask_sock
pip install flask-cors

/////////////////////////////////
// RUN THE SERVER
/////////////////////////////////

flask --app main run

/////////////////////////////////
// START THE GAME
/////////////////////////////////

Open the following link in browser:
http://localhost:5000/
